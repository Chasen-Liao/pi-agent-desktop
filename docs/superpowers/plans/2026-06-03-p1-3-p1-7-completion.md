```javascript
#!/usr/bin/env node
// Resolves the runtime dep tree of `electron-updater` and emits YAML
// extraResources entries. Replaces manual package list in
// electron-builder.yml. Run automatically by `prepack` script.
//
// Markers in electron-builder.yml:
//   <<<AUTO_EXTRA_RESOURCES_BEGIN>>> ... <<<AUTO_EXTRA_RESOURCES_END>>>

const fs = require("fs");
const path = require("path");

const ROOT = path.resolve(__dirname, "..");
const YML_PATH = path.join(ROOT, "electron-builder.yml");
const ELECTRON_UPDATER_DIR = path.join(ROOT, "node_modules", "electron-updater");
const MAX_DEPTH = 5;

function readDeps(pkgPath, seen, depth = 0) {
  if (depth > MAX_DEPTH || !fs.existsSync(pkgPath)) return;
  const pkg = JSON.parse(fs.readFileSync(pkgPath, "utf8"));
  // Resolve each dep relative to the parent's `node_modules/` first (covers
  // deduped nested copies like electron-updater/node_modules/fs-extra@10.1.0),
  // then fall back to the top-level node_modules for hoisted deps.
  const parentDir = path.dirname(path.dirname(pkgPath));
  for (const [name, ver] of Object.entries(pkg.dependencies ?? {})) {
    if (seen.has(name)) continue;
    seen.add(name);
    const nested = path.join(parentDir, "node_modules", name, "package.json");
    const topLevel = path.join(ROOT, "node_modules", name, "package.json");
    const childPath = fs.existsSync(nested) ? nested : topLevel;
    readDeps(childPath, seen, depth + 1);
  }
}

function buildEntries() {
  // Track every name that has a nested copy under electron-updater's own
  // node_modules/. These are auto-included by the electron-updater entry
  // and must NOT get a top-level extraResources entry, otherwise the
  // hoisted (often older) top-level version would shadow the nested one
  // that electron-updater actually resolves.
  const nestedOnly = new Set();
  const topLevel = new Set();
  for (const name of fs.readdirSync(path.join(ELECTRON_UPDATER_DIR, "node_modules"))) {
    if (name === ".bin") continue;
    if (fs.existsSync(path.join(ELECTRON_UPDATER_DIR, "node_modules", name, "package.json"))) {
      nestedOnly.add(name);
    }
  }

  // Walk the declared dep tree; collect top-level installed deps.
  const seen = new Set();
  readDeps(path.join(ELECTRON_UPDATER_DIR, "package.json"), seen, 0);
  for (const name of seen) {
    if (nestedOnly.has(name)) continue; // covered by electron-updater copy
    if (fs.existsSync(path.join(ROOT, "node_modules", name, "package.json"))) {
      topLevel.add(name);
    }
  }
  topLevel.add("electron-updater"); // always include the root

  const entries = [];
  for (const name of [...topLevel].sort()) {
    entries.push(
      `  - from: node_modules/${name}\n` +
      `    to: app/node_modules/${name}`
    );
  }
  return entries.join("\n");
}

function syncYml() {
  const yml = fs.readFileSync(YML_PATH, "utf8");
  const begin = "<<<AUTO_EXTRA_RESOURCES_BEGIN>>>";
  const end = "<<<AUTO_EXTRA_RESOURCES_END>>>";
  if (!yml.includes(begin) || !yml.includes(end)) {
    throw new Error(`Missing markers ${begin} / ${end} in electron-builder.yml`);
  }
  const block = buildEntries();
  const updated = yml.replace(
    new RegExp(`${begin}[\\s\\S]*?${end}`),
    `${begin}\n${block}\n    ${end}`
  );
  fs.writeFileSync(YML_PATH, updated);
  const lines = block.split("\n").length;
  console.log(`[sync-electron-deps] wrote ${lines} extraResources entries`);
}

if (require.main === module) {
  syncYml();
}

module.exports = { buildEntries, readDeps };
```

```typescript
      fetchContent(filePath, true).then(() => {
        // Restore after the next paint; the virtualizer may not have
        // measured the new total height yet.
        requestAnimationFrame(() => {
          if (scrollRef.current) {
            scrollRef.current.scrollTop = savedScrollTop;
          }
        });
      });
    });
```

注意：仅在 `isLargeSource` 期间 `scrollRef.current` 才指向虚拟器容器；小文件 `scrollRef.current` 是 `null`（容器没渲染）。`?.scrollTop ?? 0` 安全降级。
